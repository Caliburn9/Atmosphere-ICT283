32 /**
33 * @brief Sorts a vector using the Merge Sort algorithm.
34 *
35 * This recursive function implements the Merge Sort algorithm to sort
36 * elements in a vector between the given start and end indices.
37 *
38 * @tparam T The type of elements in the vector.
39 * @param data The vector to be sorted.
40 * @param start The starting index of the segment to sort.
41 * @param end The ending index of the segment to sort.
42 * @pre data must be a valid Vector<T> with elements at indices from start to end.
43 * @post The segment data[start...end] is sorted in ascending order.
44 */
45template <class T>
46void MergeSort(Vector<T> & data, int start, int end)
47{
48 if (start >= end)
49 {
50 return;
51 }
52
53 if (end - start == 1)
54 {
55 if (data[start] > data[end])
56 {
57 Swap(data[start], data[end]);
58 }
59 return;
60 }
61
62 int mid = (start + end) / 2;
63
64 MergeSort(data, start, mid);
65 MergeSort(data, mid + 1, end);
66
67 Merge(data, start, mid, end);
68}
