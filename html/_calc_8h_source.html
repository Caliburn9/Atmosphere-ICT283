12 /**
13 * @brief Calculates the mean (average) of a vector of values.
14 *
15 * This templated function takes in a vector of any numeric type and calculates the
16 * mean by summing all elements and dividing by the total number.
17 *
18 * @tparam T - The type of the elements stored in the vector (for example float or int).
19 * @param vec - A vector containing elements of type T.
20 * @param total - The number of elements to include in the mean calculation.
21 * @return The mean of the elements in the vector.
22 * @pre total > 0 and all elements in vec must be valid numeric values.
23 * @post Returns the calculated mean of type T.
24 */
25template <class T>
26T CalculateMean(const Vector<T> & vec, int total)
27{
28 T sum = 0;
29 for (int i = 0; i < total; i++)
30 {
31 sum += vec[i];
32 }
33 return sum/total;

38 /**
39 * @brief Calculates the sample standard deviation of a vector of values.
40 *
41 * This templated function uses the sample standard deviation formula, where the
42 * variance is divided by (N - 1). It assumes a precalculated mean is provided.
43 *
44 * @tparam T - The type of the elements stored in the vector (e.g., float, int).
45 * @param vec - A vector containing elements of type T.
46 * @param N - The number of elements to include in the standard deviation calculation.
47 * @param mean - The precalculated mean of the vector elements.
48 * @return The sample standard deviation of the elements in the vector.
49 * @pre N > 1 and all elements in vec must be valid numeric values.
50 * @post Returns the sample standard deviation.
51 */
52template <class T>
53float CalculateStandardDeviation(const Vector<T> & vec, int N, float mean)
54{
55 T sum = 0;
56 for (int i = 0; i < N; ++i)
57 {
58 T diff = vec[i] - mean;
59 sum += diff * diff;
60 }
61 return sqrt(sum / (N - 1));
62}

91 /**
92 * @brief Calculates the Sample Pearson Correlation Coefficient (sPCC) between two vectors.
93 *
94 * This function measures the linear correlation between two data sets by computing
95 * the covariance and variances, then applying the sPCC formula.
96 *
97 * @tparam T The numeric type stored in the vectors (e.g., float).
98 * @param x The first data vector.
99 * @param y The second data vector.
100 * @return The correlation coefficient between -1.0 and 1.0, or -1 if the inputs are invalid.
101 * @pre x and y must be non-empty and of equal size.
102 * @post No modification to input vectors.
103 */
104template <class T>
105float sPCC(const Vector<T> & x, const Vector<T> & y)
106{
107 if (!ValidateVectorSizes(x, y))
108 {
109 return -1;
110 }
111
112 float meanX = CalculateMean(x, x.GetSize());
113 float meanY = CalculateMean(y, y.GetSize());
114
115 float sumXY = CalculateCovariance(x, y, meanX, meanY);
116 float sumX2 = CalculateVariance(x, meanX);
117 float sumY2 = CalculateVariance(y, meanY);
118
119 float sqrtX2 = sqrt(sumX2);
120 float sqrtY2 = sqrt(sumY2);
121
122 float denom = sqrtX2 * sqrtY2;
123 if (denom == 0)
124 {
125 return 0;
126 }
127
128 return sumXY / denom;
129}

157 /**
158 * @brief Computes the covariance component for sPCC between two vectors.
159 *
160 * @tparam T The numeric type stored in the vectors.
161 * @param x The first data vector.
162 * @param y The second data vector.
163 * @param meanX The precomputed mean of vector x.
164 * @param meanY The precomputed mean of vector y.
165 * @return The sum of products of differences from the means.
166 */
167template <class T>
168float CalculateCovariance(const Vector<T>& x, const Vector<T>& y, float meanX, float meanY) {
169 float sumXY = 0;
170 for (int i = 0; i < x.GetSize(); i++) {
171 float xDiff = x[i] - meanX;
172 float yDiff = y[i] - meanY;
173 sumXY += xDiff * yDiff;
174 }
175 return sumXY;
176}

182 *
183 * This function computes the variance numerator (∑(x - mean)²), useful for correlation or standard deviation.
184 *
185 * @tparam T The numeric type of the vector elements.
186 * @param vec The data vector.

188 * @return The sum of squared differences from the mean.

190template <class T>

203 * @brief Calculates the Mean Absolute Deviation (MAD) of a vector of values.
204 *
205 * This templated function computes the Mean Absolute Deviation (MAD), which is
206 * the average of the absolute differences between each value and the mean.
207 * It is a measure of dispersion in the dataset and is less sensitive to outliers than standard deviation.
208 *
209 * @tparam T - The type of the elements stored in the vector (e.g., float, int).
210 * @param vec - A vector containing elements of type T.
211 * @param size - The number of elements to include in the MAD calculation.
212 * @return The mean absolute deviation of the elements in the vector.
213 * @pre size > 0 and all elements in vec must be valid numeric values.
214 * @post Returns the mean absolute deviation as a float.
215 */
216template <class T>
217float MAD(const Vector<T> & vec, int size)
218{
219 T sum = 0;
220 float mean = CalculateMean(vec, size);
221 for (int i = 0; i < size; i++)
222 {
223 sum += std::abs(vec[i] - mean);
224 }
225 return sum / size;
226}
