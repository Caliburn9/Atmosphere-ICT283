39 *
40 * This templated function uses the sample standard deviation formula, where the
41 * variance is divided by (N - 1). It assumes a precalculated mean is provided.
42 *
43 * @tparam T - The type of the elements stored in the vector (e.g., float, int).
44 * @param vec - A vector containing elements of type T.
45 * @param N - The number of elements to include in the standard deviation calculation.
46 * @param mean - The precalculated mean of the vector elements.
47 * @return The sample standard deviation of the elements in the vector.
48 * @pre N > 1 and all elements in vec must be valid numeric values.
49 * @post Returns the sample standard deviation of type T.
50 */
51template <class T>
52T CalculateStandardDeviation(const Vector<T> & vec, int N, float mean)
53{
54 T sum = 0;
55 for (int i = 0; i < N; ++i)
56 {
57 T diff = vec[i] - mean;
58 sum += diff * diff;
59 }
60 return static_cast<T>(sqrt(sum / (N - 1)));
