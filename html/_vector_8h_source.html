261template <class T>
262void Vector<T>::PushBack(const T& val)
263{
264 if (m_capacity == 0)
265 {
266 m_capacity = 1;
267 m_data = new T[m_capacity];
268 }
269
270 m_data[m_size] = val;
271 m_size++;
272
273 if (m_size > (m_capacity / 2))
274 {
275 int newCapacity = m_capacity + m_size + 1;
276 T* newData = new T[newCapacity];
277 for (int i = 0; i < m_size; i++)
278 {
279 newData[i] = m_data[i];
280 }
281 delete[] m_data;
282 m_data = newData;
283 m_capacity = newCapacity;
284 }
285}
