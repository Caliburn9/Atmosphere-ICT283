11 /**
12 * @class Collector
13 * @brief A static utility class for collecting and storing atmospheric data measurements by month.
14 *
15 * The Collector class provides a centralized mechanism for gathering atmospheric measurements
16 * (wind speed, air temperature, and solar radiation) from AtmosRecType records. It maintains three static vectors to
17 * store the collected data and allows filtering by a target month. The class offers separate collection
18 * methods for different measurement combinations and provides access to the stored data vectors.
19 *
20 * @author Nabeel
21 * @version 01
22 * @date 22/07/2025 Nabeel, Started
23 *
24 * @todo Nothing
25 *
26 * @bug No bugs so far
27 */
28
29class Collector {

31 /**
32 * @brief Clears all stored data and resets the target month.
33 *
34 * Empties the vectors holding wind speed, temperature, and solar radiation values,
35 * and resets the target month to 0.
36 *
37 * @return void
38 * @pre None.
39 * @post All vectors are cleared and m_targetMonth is reset to 0.
40 */
41 static void Clear();
42
43 /**
44 * @brief Sets the month used as a filter for data collection.
45 *
46 * @param month Integer representing the target month (1-12).
47 * @return void
48 * @pre 1 <= month <= 12.
49 * @post m_targetMonth is updated to the provided value.
50 */
51 static void SetTargetMonth(int month);
52
53 /**
54 * @brief Retrieves the current target month.
55 *
56 * @return Integer value of the currently set target month.
57 * @pre m_targetMonth must have been set via SetTargetMonth().
58 * @post No changes to internal state.

101 /**

106 *
107 * @param a The AtmosRecType record to extract data from.
108 * @pre m_targetMonth must be set; a must be valid.
109 * @post If conditions are met, data is appended to m_sVec and m_rVec.

111 static void CollectSR(const AtmosRecType & a);

113 /**

118 *
119 * @param a The AtmosRecType record to extract data from.
120 * @pre m_targetMonth must be set; a must be valid.
121 * @post If conditions are met, data is appended to m_tVec and m_rVec.
122 */
123 static void CollectTR(const AtmosRecType & a);
124
125private:
